54 // Utility template function used to print container
55 template<class T> void PrintContainer(T first, T last)
56 {
57  if (first == last)
58  cout << "<EMPTY SET>";
59  else
60  for(;first != last; ++first)
61  cout << *first << ";";
62  cout << endl;
63 }

65 int main(void)
66 {
67  try
68  {
69  bm::bvector<> bv;
70 
71  bv[10] = true;
72  bv[100] = true;
73  bv[10000] = true;
74 
75  cout << "Source set:";
76  PrintContainer(bv.first(), bv.end());
77 
78  // copy all bitset information into STL vector using copy algorithm
79  {
80  vector<unsigned> vect;
81  vect.resize(bv.count());
82  std::copy(bv.first(), bv.end(), vect.begin());
83  cout << "Vector:";
84  PrintContainer(vect.begin(), vect.end());
85  }
86 
87  // doing the same with the help of back_inserter
88 
89  {
90  list<unsigned> lst;
91  std::copy(bv.first(), bv.end(), std::back_inserter(lst));
92  cout << "List:";
93  PrintContainer(lst.begin(), lst.end());
94  }
95 
96  {
97  vector<unsigned> vect;
98  vector<unsigned> res1, res2, res3;
99 
100  vect.push_back(100);
101  vect.push_back(15);
102  vect.push_back(150);
103 
104  cout << "Argument vector for set operations:";
105  PrintContainer(vect.begin(), vect.end());
106 
107  // set should be ordered by < to make set algorithms possible
108  std::sort(vect.begin(), vect.end());
109  cout << endl;
110 
111  std::set_union(bv.first(), bv.end(),
112  vect.begin(), vect.end(),
113  std::back_inserter(res1)); //10;15;100;150;10000
114  cout << "Set union:" << endl;
115  PrintContainer(res1.begin(), res1.end());
116 
117  std::set_intersection(bv.first(), bv.end(),
118  vect.begin(), vect.end(),
119  std::back_inserter(res2)); // 100
120  cout << "Set intersection:" << endl;
121  PrintContainer(res2.begin(), res2.end());
122 
123  vector<unsigned>::const_iterator it1 = vect.begin();
124  vector<unsigned>::const_iterator it2 = vect.end();
125  bm::bvector<>::enumerator en = bv.first();
126  bm::bvector<>::enumerator en2= bv.end();
127 
128  std::set_difference(en, en2,
129  it1, it2,
130  std::back_inserter(res3)); // 10;10000
131 
132  cout << "Set diff:" << endl;
133  PrintContainer(res3.begin(), res3.end());
134 
135  }
