45 typedef bm::bvector<> bvector_type;
46 
47 // define the sparse vector type for 'char' type using bvector as
48 // a container of bits for bit-transposed planes
49 // 32 - is maximum string length for this container.
50 // Memory allocation is dynamic using sparse techniques, so this number
51 // just defines the max capacity.
52 //
53 typedef bm::str_sparse_vector<char, bvector_type, 32> str_sv_type;

56 // generate collection of strings from integers and shuffle it
57 //
58 static
59 void generate_string_set(vector<string>& str_vec)
60 {
61  const unsigned max_coll = 50000;
62 
63  str_vec.resize(0);
64  string str;
65  for (unsigned i = 10; i < max_coll; i += rand() % 3)
66  {
67  str = to_string(i);
68  str_vec.emplace_back(str);
69  } // for i
70 
71  // shuffle the data set
72  //
73  std::random_device rd;
74  std::mt19937 g(rd());
75  std::shuffle(str_vec.begin(), str_vec.end(), g);
76 }

79 int main(void)
80 {
81  try
82  {
83  str_sv_type str_sv;
84 
85  vector<string> str_vec;
86  generate_string_set(str_vec);
87  std::sort(str_vec.begin(), str_vec.end()); // sort the input vector
88 
89  // load sparse vector from an STL container
90  //
91  {
92  str_sv_type str_sv_tmp; // temp vector
93  {
94  str_sv_type::back_insert_iterator bi =
95  str_sv_tmp.get_back_inserter();
96  for (auto str : str_vec)
97  {
98  bi = str;
99  }
100  // it is important to use flush, because back inserter is
101  // buffering data. Of cause it flashes automatically on
102  // destruction but explicit flush is somewhat better
103  // because of possible exception is thrown here and not from
104  // destructor
105  //
106 
107  bi.flush();
108  }
109 
110  // final step is re-mapping, which increses chances for
111  // good memory compression
112  //
113  str_sv.remap_from(str_sv_tmp);
114 
115  BM_DECLARE_TEMP_BLOCK(tb)
116  str_sv.optimize(tb); // optimize the vector
117  }
118 
119  // serialize and save
120  //
121  {
122  std::string fname = "test.sv";
123  bm::sparse_vector_serial_layout<str_sv_type> sv_lay;
124 
125  BM_DECLARE_TEMP_BLOCK(tb)
126  bm::sparse_vector_serialize(str_sv, sv_lay, tb);
127 
128  std::ofstream fout(fname.c_str(), std::ios::binary);
129  if (!fout.good())
130  {
131  return -1;
132  }
133  const char* buf = (char*)sv_lay.buf();
134  fout.write(buf, (unsigned)sv_lay.size());
135  if (!fout.good())
136  {
137  return -1;
138  }
139  fout.close();
140  }
141 
142  }
143  catch(std::exception& ex)

146  return 1;
147  }
148 
149

151 }
