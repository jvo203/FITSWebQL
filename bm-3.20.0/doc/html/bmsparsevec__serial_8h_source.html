124 /*!
125  \brief Serialize sparse vector into a memory buffer(s) structure
126 
127  Serialization format:
128 
129  | HEADER | BIT-VECTORS ... | REMAP_MATRIX
130 
131  Header structure:
132  -----------------
133  BYTE+BYTE: Magic-signature 'BM' or 'BC' (c-compressed)
134  BYTE : Byte order ( 0 - Big Endian, 1 - Little Endian)
135  {
136  BYTE : Number of Bit-vector plains (total) (non-zero when < 255 plains)
137  |
138  BYTE: zero - flag of large plain matrix
139  INT64: Nnmber of bit-vector plains
140  }
141  INT64: Vector size
142  INT64: Offset of plain 0 from the header start (value 0 means plain is empty)
143  INT64: Offset of plain 1 from
144  ...
145  INT32: reserved
146 
147 Bit-vectors:
148 ------------
149  Based on current bit-vector serialization
150 
151 Remap Matrix:
152  SubHeader | Matrix BLOB
153 
154  sub-header:
155  BYTE: 'R' (remapping) or 'N' (no remapping)
156  N - means no other info is saved on the stream
157  INT64: remap matrix size
158 
159  \ingroup svserial
160 */

313 /**
314  \brief Serialize compressed collection into memory buffer
315 
316 Serialization format:
317 
318 
319 <pre>
320  | MAGIC_HEADER | ADDRESS_BITVECTROR | LIST_OF_BUFFER_SIZES | BUFFER(s)
321 
322  MAGIC_HEADER:
323  BYTE+BYTE: Magic-signature 'BM' or 'BC'
324  BYTE : Byte order ( 0 - Big Endian, 1 - Little Endian)
325 
326  ADDRESS_BITVECTROR:
327  INT64: address bit-vector size
328  <memblock>: serialized address bit-vector
329 
330  LIST_OF_BUFFER_SIZES:
331  INT64 - buffer sizes count
332  INT32 - buffer size 0
333  INT32 - buffer size 1
334  ...
335 
336  BUFFERS:
337  <memblock>: block0
338  <memblock>: block1
339  ...
340 
341 </pre>
342 */
343 
344 template<class CBC>
345 void compressed_collection_serializer<CBC>::serialize(const CBC& buffer_coll,
346  buffer_type& buf,
347  bm::word_t* temp_block)
348 {
349  statistics_type st;
350  buffer_coll.calc_stat(&st);
351 
352  buf.resize(st.max_serialize_mem);
353 
354  // ptr where bit-plains start
355  unsigned char* buf_ptr = buf.data();
356 
357  bm::encoder enc(buf.data(), buf.capacity());
358  ByteOrder bo = globals<true>::byte_order();
359  enc.put_8('B');
360  enc.put_8('C');
361  enc.put_8((unsigned char)bo);
362 
363  unsigned char* mbuf1 = enc.get_pos(); // bookmark position
364  enc.put_64(0); // address vector size (reservation)
365 
366  buf_ptr = enc.get_pos();
367 
368  const address_resolver_type& addr_res = buffer_coll.resolver();
369  const bvector_type& bv = addr_res.get_bvector();
370  {
371  bm::serializer<bvector_type > bvs(temp_block);
372  bvs.gap_length_serialization(false);
373  bvs.set_compression_level(4);
374 
375  size_t addr_bv_size = bvs.serialize(bv, buf_ptr, buf.size());
376  buf_ptr += addr_bv_size;
377 
378  enc.set_pos(mbuf1); // rewind to bookmark
379  enc.put_64(addr_bv_size); // save the address vector size
380  }
381  enc.set_pos(buf_ptr); // restore stream position
382  size_t coll_size = buffer_coll.size();
383 
384  enc.put_64(coll_size);
385 
386  // pass 1 (save buffer sizes)
387  {
388  for (unsigned i = 0; i < buffer_coll.size(); ++i)
389  {
390  const buffer_type& cbuf = buffer_coll.get(i);
391  unsigned sz = (unsigned)cbuf.size();
392  enc.put_32(sz);
393  } // for i
394  }
395  // pass 2 (save buffers)
396  {
397  for (unsigned i = 0; i < buffer_coll.size(); ++i)
398  {
399  const buffer_type& cbuf = buffer_coll.get(i);
400  unsigned sz = (unsigned)cbuf.size();
401  enc.memcpy(cbuf.buf(), sz);
402  } // for i
403  }
404  buf.resize(enc.size());
405 
406 }

409 template<class CBC>
410 int compressed_collection_deserializer<CBC>::deserialize(
411  CBC& buffer_coll,
412  const unsigned char* buf,
413  bm::word_t* temp_block)
414 {
415  // TODO: implement correct processing of byte-order corect deserialization
416  // ByteOrder bo_current = globals<true>::byte_order();
417 
418  bm::decoder dec(buf);
419  unsigned char h1 = dec.get_8();
420  unsigned char h2 = dec.get_8();
421 
422  BM_ASSERT(h1 == 'B' && h2 == 'C');
423  if (h1 != 'B' && h2 != 'C') // no magic header? issue...
424  {
425  return -1;
426  }
427  //unsigned char bv_bo =
428  dec.get_8();
429 
430  // -----------------------------------------
431  // restore address resolver
432  //
433  bm::id64_t addr_bv_size = dec.get_64();
434 
435  const unsigned char* bv_buf_ptr = dec.get_pos();
436 
437  address_resolver_type& addr_res = buffer_coll.resolver();
438  bvector_type& bv = addr_res.get_bvector();
439  bv.clear();
440 
441  bm::deserialize(bv, bv_buf_ptr, temp_block);
442  addr_res.sync();
443 
444  unsigned addr_cnt = bv.count();
445 
446  dec.seek((int)addr_bv_size);
447 
448  // -----------------------------------------
449  // read buffer sizes
450  //
451  bm::id64_t coll_size = dec.get_64();
452  if (coll_size != addr_cnt)
453  {
454  return -2; // buffer size collection does not match address vector
455  }
456 
457  typedef std::vector<unsigned>::size_type vect_size_type;
458  std::vector<unsigned> buf_size_vec;
459  buf_size_vec.resize(vect_size_type(coll_size));
460  {
461  for (unsigned i = 0; i < coll_size; ++i)
462  {
463  unsigned sz = dec.get_32();
464  buf_size_vec[i] = sz;
465  } // for i
466  }
467 
468  {
469  container_type& buf_vect = buffer_coll.container();
470  buf_vect.resize(vect_size_type(coll_size));
471  for (unsigned i = 0; i < coll_size; ++i)
472  {
473  unsigned sz = buf_size_vec[i];
474  buffer_type& b = buf_vect.at(i);
475  b.resize(sz);
476  dec.memcpy(b.data(), sz);
477  } // for i
478  }
479 
480  buffer_coll.sync();
481 
482  return 0;
483 }

498 template<typename SV>
499 void sparse_vector_serializer<SV>::serialize(const SV& sv,
500  sparse_vector_serial_layout<SV>& sv_layout)
501 {
502 // const unsigned max_v_plains = 255; // legacy...
503 
504  typename SV::statistics sv_stat;
505  sv.calc_stat(&sv_stat);
506  unsigned char* buf = sv_layout.reserve(sv_stat.max_serialize_mem);
507 
508  bm::encoder enc(buf, (unsigned)sv_layout.capacity());
509  unsigned plains = sv.stored_plains();
510 
511  // header size in bytes
512  unsigned h_size = 1 + 1 + // "BM" or "BC" (magic header)
513  1 + // byte-order
514  1 + // number of bit-plains (for vector)
515  8 + // size (internal 64-bit)
516  (8 * plains) + // offsets of all plains
517  4; // reserve
518  // for large plain matrixes
519  {
520  h_size += 1 + // version number
521  8; // number of plains (64-bit)
522  }
523 
524  // -----------------------------------------------------
525  // Serialize all bvector plains
526  //
527 
528  unsigned char* buf_ptr = buf + h_size; // ptr where plains start (start+hdr)
529 
530  unsigned i;
531  for (i = 0; i < plains; ++i)
532  {
533  typename SV::bvector_type_const_ptr bv = sv.get_plain(i);
534  if (!bv) // empty plain
535  {
536  sv_layout.set_plain(i, 0, 0);
537  continue;
538  }
539 
540  unsigned buf_size =
541  bvs_.serialize(*bv, buf_ptr, sv_stat.max_serialize_mem);
542 
543  sv_layout.set_plain(i, buf_ptr, buf_size);
544  buf_ptr += buf_size;
545  if (sv_stat.max_serialize_mem > buf_size)
546  {
547  sv_stat.max_serialize_mem -= buf_size;
548  }
549  else
550  {
551  BM_ASSERT(0); // TODO: throw an exception here
552  }
553  } // for i
554 
555  // -----------------------------------------------------
556  // serialize the re-map matrix
557  //
558  if (bm::conditional<SV::is_remap_support::value>::test()) // test remapping trait
559  {
560  bm::encoder enc_m(buf_ptr, sv_stat.max_serialize_mem);
561  if (sv.is_remap())
562  {
563  bm::id64_t remap_size = sv.remap_size();
564  const unsigned char* matrix_buf = sv.get_remap_buffer();
565  BM_ASSERT(matrix_buf);
566  BM_ASSERT(remap_size);
567 
568  enc_m.put_8('R');
569  enc_m.put_64(remap_size);
570  enc_m.memcpy(matrix_buf, remap_size);
571  enc_m.put_8('E'); // end of matrix (integrity check token)
572  }
573  else
574  {
575  enc_m.put_8('N');
576  }
577  buf_ptr += enc_m.size(); // add mattrix encoded data size
578  }
579 
580  sv_layout.resize(size_t(buf_ptr - buf)); // set the true occupied size
581 
582  // -----------------------------------------------------
583  // save the header
584  //
585  ByteOrder bo = bm::globals<true>::byte_order();
586 
587  enc.put_8('B'); // magic header 'BM' - bit matrix 'BC' - bit compressed
588  if (sv.is_compressed())
589  enc.put_8('C');
590  else
591  enc.put_8('M');
592 
593  enc.put_8((unsigned char)bo); // byte order
594 
595  enc.put_8(0); // number of plains == 0 (legacy magic number)
596  enc.put_8(1); // matrix serialization version
597  enc.put_64(plains); // number of rows in the bit-matrix
598 
599  enc.put_64(sv.size_internal());
600 
601  // save the offset table (part of the header)
602  //
603  for (i = 0; i < plains; ++i)
604  {
605  const unsigned char* p = sv_layout.get_plain(i);
606  if (!p)
607  {
608  enc.put_64(0);
609  continue;
610  }
611  size_t offset = size_t(p - buf);
612  enc.put_64(offset);
613  } // for
614 }

627 template<typename SV>
628 void sparse_vector_deserializer<SV>::deserialize(SV& sv,
629  const unsigned char* buf)
630 {
631  // TODO: implement correct processing of byte-order corect deserialization
632  // ByteOrder bo_current = globals<true>::byte_order();
633 
634  bm::decoder dec(buf);
635  unsigned char h1 = dec.get_8();
636  unsigned char h2 = dec.get_8();
637 
638  BM_ASSERT(h1 == 'B' && (h2 == 'M' || h2 == 'C'));
639 
640  if (h1 != 'B' && (h2 != 'M' || h2 != 'C')) // no magic header?
641  {
642  #ifndef BM_NO_STL
643  throw std::logic_error("Invalid serialization signature header");
644  #else
645  BM_THROW(BM_ERR_SERIALFORMAT);
646  #endif
647  }
648  unsigned char matr_s_ser = 0;
649  //unsigned char bv_bo =
650  dec.get_8();
651  unsigned plains = dec.get_8();
652  if (plains == 0) // bit-matrix
653  {
654  matr_s_ser = dec.get_8(); // matrix serialization version
655  plains = (unsigned) dec.get_64(); // number of rows in the bit-matrix
656  }
657 
658  unsigned sv_plains = sv.stored_plains();
659  if (!plains || plains > sv_plains)
660  {
661  #ifndef BM_NO_STL
662  throw std::logic_error("Invalid serialization target (bit depth)");
663  #else
664  BM_THROW(BM_ERR_SERIALFORMAT);
665  #endif
666  }
667 
668  sv.clear();
669 
670  bm::id64_t sv_size = dec.get_64();
671  if (sv_size == 0)
672  return; // empty vector
673 
674  sv.resize_internal((unsigned)sv_size);
675  bm::word_t* temp_block = 0;
676 
677  const unsigned char* remap_buf_ptr = 0;
678 
679  unsigned i = 0;
680  for (i = 0; i < plains; ++i)
681  {
682  size_t offset = (size_t) dec.get_64();
683  if (offset == 0) // null vector
684  {
685  continue;
686  }
687  const unsigned char* bv_buf_ptr = buf + offset; // seek to position
688  bvector_type* bv = sv.get_plain(i);
689  BM_ASSERT(bv);
690  if (!temp_block)
691  {
692  typename bvector_type::blocks_manager_type& bv_bm =
693  bv->get_blocks_manager();
694  temp_block = bv_bm.check_allocate_tempblock();
695  }
696  unsigned read_bytes = deserial_.deserialize(*bv, bv_buf_ptr, temp_block);
697  remap_buf_ptr = bv_buf_ptr + read_bytes;
698  } // for i
699 
700  // ----------------------------
701  // load the remap matrix
702  //
703  if (bm::conditional<SV::is_remap_support::value>::test()) // test remapping trait
704  {
705  if (matr_s_ser > 0)
706  {
707  if (remap_buf_ptr)
708  {
709  bm::decoder dec_m(remap_buf_ptr);
710  unsigned char rh = dec_m.get_8();
711  if (rh == 'N') // no remap matrix here
712  {
713  }
714  else
715  if (rh == 'R')
716  {
717  size_t remap_size = (size_t) dec_m.get_64();
718  unsigned char* remap_buf = sv.init_remap_buffer();
719  BM_ASSERT(remap_buf);
720  size_t target_remap_size = sv.remap_size();
721  if (!remap_size || !remap_buf || remap_size != target_remap_size)
722  {
723  #ifndef BM_NO_STL
724  throw std::logic_error("Invalid serialization format (remap size)");
725  #else
726  BM_THROW(BM_ERR_SERIALFORMAT);
727  #endif
728  }
729  dec_m.memcpy(remap_buf, remap_size);
730  unsigned char end_tok = dec_m.get_8();
731  if (end_tok != 'E')
732  {
733  #ifndef BM_NO_STL
734  throw std::logic_error("Invalid serialization format");
735  #else
736  BM_THROW(BM_ERR_SERIALFORMAT);
737  #endif
738  }
739  sv.set_remap();
740  }
741  else // unknown serialization token
742  {
743  #ifndef BM_NO_STL
744  throw std::logic_error("Invalid serialization format (remap error)");
745  #else
746  BM_THROW(BM_ERR_SERIALFORMAT);
747  #endif
748  }
749 
750  }
751  }
752  } // if remap traits
753 
754  sv.sync(true); // force sync, recalculate RS index, remap tables, etc
