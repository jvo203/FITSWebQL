35 // Custom allocator keeps number of all alloc/free calls.
36 // It also reservs the front word of the allocated block and saves
37 // number of elements allocated. On deallocation it makes sure
38 // it deallocates the same size as allocated
39 //
40 // Please note, that this sample allocator is NOT compatible with SIMD
41 // optimizations, requiring special address alignment
42 //
43 
44 class dbg_block_allocator
45 {
46 public:
47 static unsigned na_;
48 static unsigned nf_;
49 
50  static bm::word_t* allocate(size_t n, const void *)
51  {
52  ++na_;
53  assert(n);
54  bm::word_t* p =
55  (bm::word_t*) ::malloc((n+1) * sizeof(bm::word_t));
56  *p = (unsigned)n;
57  return ++p;
58  }
59 
60  static void deallocate(bm::word_t* p, size_t /* n */)
61  {
62  ++nf_;
63  --p;
64  ::free(p);
65  }
66 
67  static int balance()
68  {
69  return int(nf_ - na_);
70  }
71 };
72

74 unsigned dbg_block_allocator::nf_ = 0;

76 class dbg_ptr_allocator
77 {
78 public:
79 static unsigned na_;
80 static unsigned nf_;
81 
82  static void* allocate(size_t n, const void *)
83  {
84  ++na_;
85  assert(sizeof(size_t) == sizeof(void*));
86  void* p = ::malloc((n+1) * sizeof(void*));
87  size_t* s = (size_t*) p;
88  *s = n;
89  return (void*)++s;
90  }
91 
92  static void deallocate(void* p, size_t /* n */)
93  {
94  ++nf_;
95  size_t* s = (size_t*) p;
96  --s;
97  ::free(s);
98  }
99 
100  static int balance()
101  {
102  return int(nf_ - na_);
103  }
104 
105 };
