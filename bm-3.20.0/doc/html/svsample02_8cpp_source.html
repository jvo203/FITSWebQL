44 int main(void)

54 
55  for (unsigned i = 0; i < 128000; ++i)

57  sv1.push_back(8);

61  sv1.optimize(tb);

63  bm::sparse_vector_serial_layout<svector> sv_lay;
64  bm::sparse_vector_serialize(sv1, sv_lay, tb);
65 
66  // copy serialization buffer to some other location

68  //
69  const unsigned char* buf = sv_lay.buf();

71

73  ::memcpy(&tmp_buf[0], buf, buf_size);
74 
75  int res = bm::sparse_vector_deserialize(sv2, &tmp_buf[0], tb);
76  if (res != 0)
