159 public:
160  bit_out(TEncoder& dest)
161  : dest_(dest), used_bits_(0), accum_(0)

164  ~bit_out()
165  {
166  if (used_bits_)

170  void put_bit(unsigned value)
171  {
172  BM_ASSERT(value <= 1);
173  accum_ |= (value << used_bits_);
174  if (++used_bits_ == (sizeof(accum_) * 8))
175  flush_accum();
176  }
177 
178  void put_bits(unsigned value, unsigned count)
179  {
180  unsigned used = used_bits_;
181  unsigned acc = accum_;
182 
183  {
184  unsigned mask = ~0u;
185  mask >>= (sizeof(accum_) * 8) - count;
186  value &= mask;
187  }
188  for (;count;)
189  {
190  unsigned free_bits = unsigned(sizeof(accum_) * 8) - used;
191  BM_ASSERT(free_bits);
192  acc |= value << used;
193 
194  if (count <= free_bits)
195  {
196  used += count;
197  break;
198  }
199  else
200  {
201  value >>= free_bits;
202  count -= free_bits;
203  dest_.put_32(acc);
204  acc = used = 0;
205  continue;
206  }
207  }
208  if (used == (sizeof(accum_) * 8))
209  {
210  dest_.put_32(acc);
211  acc = used = 0;
212  }
213  used_bits_ = used;
214  accum_ = acc;
215  }
216 
217  void put_zero_bit()
218  {
219  if (++used_bits_ == (sizeof(accum_) * 8))
220  flush_accum();
221  }
222 
223  void put_zero_bits(unsigned count)
224  {
225  unsigned used = used_bits_;
226  unsigned free_bits = (sizeof(accum_) * 8) - used;
227  if (count >= free_bits)
228  {
229  flush_accum();
230  count -= free_bits;
231  used = 0;
232 
233  for ( ;count >= sizeof(accum_) * 8; count -= sizeof(accum_) * 8)
234  {
235  dest_.put_32(0);
236  }
237  used += count;
238  }
239  else
240  {
241  used += count;
242  }
243  accum_ |= (1 << used);
244  if (++used == (sizeof(accum_) * 8))
245  flush_accum();
246  else
247  used_bits_ = used;
248  }

251  void gamma(unsigned value)
252  {
253  BM_ASSERT(value);
254 
255  unsigned logv = bm::bit_scan_reverse32(value);
256 
257  // Put zeroes + 1 bit
258 
259  unsigned used = used_bits_;
260  unsigned acc = accum_;
261  const unsigned acc_bits = (sizeof(acc) * 8);
262  unsigned free_bits = acc_bits - used;
263 
264  {
265  unsigned count = logv;
266  if (count >= free_bits)
267  {
268  dest_.put_32(acc);
269  acc = used ^= used;
270  count -= free_bits;
271 
272  for ( ;count >= acc_bits; count -= acc_bits)
273  {
274  dest_.put_32(0);
275  }
276  used += count;
277  }
278  else
279  {
280  used += count;
281  }
282  acc |= (1 << used);
283  if (++used == acc_bits)
284  {
285  dest_.put_32(acc);
286  acc = used ^= used;
287  }
288  }
289 
290  // Put the value bits
291  //
292  {
293  unsigned mask = (~0u);
294  mask >>= acc_bits - logv;
295  value &= mask;
296  }
297  for (;logv;)
298  {
299  acc |= value << used;
300  free_bits = acc_bits - used;
301  if (logv <= free_bits)
302  {
303  used += logv;
304  break;
305  }
306  else
307  {
308  value >>= free_bits;
309  logv -= free_bits;
310  dest_.put_32(acc);
311  acc = used ^= used;
312  continue;
313  }
314  } // for
315 
316  used_bits_ = used;
317  accum_ = acc;
318  }
319 
320 
321  void flush()
322  {
323  if (used_bits_)
324  flush_accum();
325  }
326 
327 private:
328  void flush_accum()
329  {
330  dest_.put_32(accum_);
331  used_bits_ = accum_ = 0;
332  }
333 private:
334  bit_out(const bit_out&);
335  bit_out& operator=(const bit_out&);
336 
337 private:
338  TEncoder& dest_; ///< Bit stream target
339  unsigned used_bits_; ///< Bits used in the accumulator
340  unsigned accum_; ///< write bit accumulator
341 };

344 /**
345  Byte based reader for un-aligned bit streaming
346 
347  @ingroup gammacode
348  @sa encoder
349 */
350 template<class TDecoder>
351 class bit_in
352 {
353 public:
354  bit_in(TDecoder& decoder)
355  : src_(decoder),
356  used_bits_(unsigned(sizeof(accum_) * 8)),
357  accum_(0)
358  {
359  }
360 
361  unsigned gamma()
362  {
363  unsigned acc = accum_;
364  unsigned used = used_bits_;
365 
366  if (used == (sizeof(acc) * 8))
367  {
368  acc = src_.get_32();
369  used ^= used;
370  }
371  unsigned zero_bits = 0;
372  while (true)
373  {
374  if (acc == 0)
375  {
376  zero_bits = unsigned(zero_bits +(sizeof(acc) * 8) - used);
377  used = 0;
378  acc = src_.get_32();
379  continue;
380  }
381  unsigned first_bit_idx =
382  #if defined(BM_x86) && (defined(__GNUG__) || defined(_MSC_VER))
383  bm::bsf_asm32(acc);
384  #else
385  bm::bit_scan_fwd(acc);
386  #endif
387  acc >>= first_bit_idx;
388  zero_bits += first_bit_idx;
389  used += first_bit_idx;
390  break;
391  } // while
392 
393  // eat the border bit
394  //
395  if (used == (sizeof(acc) * 8))
396  {
397  acc = src_.get_32();
398  used = 1;
399  }
400  else
401  {
402  ++used;
403  }
404  acc >>= 1;
405 
406  // get the value
407  unsigned current;
408 
409  unsigned free_bits = unsigned((sizeof(acc) * 8) - used);
410  if (zero_bits <= free_bits)
411  {
412  take_accum:
413  current =
414  (acc & block_set_table<true>::_left[zero_bits]) | (1 << zero_bits);
415  acc >>= zero_bits;
416  used += zero_bits;
417  goto ret;
418  }
419 
420  if (used == (sizeof(acc) * 8))
421  {
422  acc = src_.get_32();
423  used ^= used;
424  goto take_accum;
425  }
426 
427  // take the part
428  current = acc;
429  // read the next word
430  acc = src_.get_32();
431  used = zero_bits - free_bits;
432  current |=
433  ((acc & block_set_table<true>::_left[used]) << free_bits) |
434  (1 << zero_bits);
435 
436  acc >>= used;
437  ret:
438  accum_ = acc;
439  used_bits_ = used;
440  return current;
441  }
442 
443  unsigned get_bits(unsigned count)
444  {
445  BM_ASSERT(count);
446 
447  unsigned value = 0;
448  unsigned free_bits = unsigned((sizeof(accum_) * 8) - used_bits_);
449  if (count <= free_bits)
450  {
451  take_accum:
452  value =
453  (accum_ & block_set_table<true>::_left[count-1]);
454  accum_ >>= count;
455  used_bits_ += count;
456  return value;
457  }
458  if (used_bits_ == (sizeof(accum_) * 8))
459  {
460  accum_ = src_.get_32();
461  used_bits_ = 0;
462  goto take_accum;
463  }
464  value = accum_;
465  accum_ = src_.get_32();
466  used_bits_ = count - free_bits;
467  value |=
468  ((accum_ & block_set_table<true>::_left[used_bits_-1]) << free_bits);
469  accum_ >>= used_bits_;
470  return value;
471  }
472

474 private:
475  bit_in(const bit_in&);
476  bit_in& operator=(const bit_in&);
477 private:
478  TDecoder& src_; ///< Source of bytes
479  unsigned used_bits_; ///< Bits used in the accumulator
480  unsigned accum_; ///< read bit accumulator
481 };
