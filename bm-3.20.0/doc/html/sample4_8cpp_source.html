44 const unsigned MAX_VALUE = 1000000;

46 // This procedure creates very dense bitvector.
47 // The resulting set will consists mostly from ON (1) bits
48 // interrupted with small gaps of 0 bits.
49 
50 static
51 void fill_bvector(bm::bvector<>* bv)
52 {
53  for (unsigned i = 0; i < MAX_VALUE; ++i)
54  {
55  if (rand() % 2500)
56  {
57  bv->set_bit(i);
58  }
59  }
60 }

75 static
76 unsigned char* serialize_bvector(bm::serializer<bm::bvector<> >& bvs,
77  bm::bvector<>& bv)
78 {
79  // It is reccomended to optimize vector before serialization.
80  BM_DECLARE_TEMP_BLOCK(tb)
81  bm::bvector<>::statistics st;
82 
83  bv.optimize(tb, bm::bvector<>::opt_compress, &st);
84 
85  cout << "Bits count:" << bv.count() << endl;
86  cout << "Bit blocks:" << st.bit_blocks << endl;
87  cout << "GAP blocks:" << st.gap_blocks << endl;
88  cout << "Memory used:"<< st.memory_used << endl;
89  cout << "Max.serialize mem.:" << st.max_serialize_mem << endl;
90 
91  // Allocate serialization buffer.
92  unsigned char* buf = new unsigned char[st.max_serialize_mem];
93 
94  // Serialization to memory.
95  unsigned len = bvs.serialize(bv, buf, st.max_serialize_mem);
96 
97 
98  cout << "Serialized size:" << len << endl << endl;
99 
100  return buf;
101 }

104 int main(void)
105 {
106  unsigned char* buf1 = 0;
107  unsigned char* buf2 = 0;
108 
109  try
110  {
111  bm::bvector<> bv1;
112  bm::bvector<> bv2;
113 
114  bv2.set_new_blocks_strat(bm::BM_GAP); // set DGAP compression mode ON
115 
116  fill_bvector(&bv1);
117  fill_bvector(&bv2);
118 
119  // Prepare a serializer class
120  // for best performance it is best to create serilizer once and reuse it
121  // (saves a lot of memory allocations)
122  //
123  bm::serializer<bm::bvector<> > bvs;
124 
125  // next settings provide lowest size
126  bvs.byte_order_serialization(false);
127  bvs.gap_length_serialization(false);
128  bvs.set_compression_level(4);
129 
130 
131  buf1 = serialize_bvector(bvs, bv1);
132  buf2 = serialize_bvector(bvs, bv2);
133 
134  // Serialized bvectors (buf1 and buf2) now ready to be
135  // saved to a database, file or send over a network.
136 
137  // ...
138 
139  // Deserialization.
140 
141  bm::bvector<> bv3;
142 
143  // As a result of desrialization bv3 will contain all bits from
144  // bv1 and bv3:
145  // bv3 = bv1 OR bv2
146 
147  bm::deserialize(bv3, buf1);
148  bm::deserialize(bv3, buf2);
149 
150  print_statistics(bv3);
151 
152  // After a complex operation we can try to optimize bv3.
153 
154  bv3.optimize();
155 
156  print_statistics(bv3);
157  }
158  catch(std::exception& ex)
159  {
160  std::cerr << ex.what() << std::endl;
161  delete [] buf1;
162  delete [] buf2;
163  return 1;
164  }
165 
166  delete [] buf1;
167  delete [] buf2;
168 
169  return 0;
170 }
