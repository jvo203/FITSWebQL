30 /*! \file rscsample01.cpp
31  \brief Example: rsc_sparse_vector<> usage
32 
33  rsc_sparse_vector<> is a sparse vector which uses bit-transposition and
34  rank-select succinct method of compression of NULL (unassigned) values.
35  Unassigned values are dropped (as transposed columns) from the bit-matrix.
36 
37  rsc_sparse_vector<> is basically a read-only structure, which can be used
38  for compact data storage and search. Random access to elements is possible
39  with a penalty of bit-vector Rank or Select operations.
40 */
41

56 {
57  std::cout << sv.size() << ": ";

60  if (show_nulls)

79 int main(void)

81  // temp buffer to avoid unnecessary re-allocations
82  BM_DECLARE_TEMP_BLOCK(tb)
83 
84  try
85  {
86  sparse_vector_u32 sv1(bm::use_null); // use null is needed to build rsc vector
87  rsc_sparse_vector_u32 csv2;
88 
89  // fill in sparse vector leaving some unassigned gaps
90  for (unsigned i = 0; i < 15; i+=3)
91  {
92  sv1[i] = i;
93  }
94 
95  print_svector(sv1); // print sparse vector disregard NULL values (printed as 0)
96  print_svector(sv1, true); // print sparse vector show NULLs (unassigned)
97 
98  csv2.load_from(sv1); // load rank-select-compacted (rsc) sparse vector
99 
100  // print results - it should look the same
101  print_svector(csv2);
102  print_svector(csv2, true);
103 
104  // serialize rsc vector
105 
106  // optimize memory allocation of sparse vector
107  csv2.optimize(tb);
108 
109  bm::sparse_vector_serial_layout<rsc_sparse_vector_u32> sv_lay;
110  bm::sparse_vector_serialize(csv2, sv_lay, tb);
111 
112  // memory copy just simulates network or DB transaction
113  const unsigned char* buf = sv_lay.buf();
114  size_t buf_size = sv_lay.size();
115 
116  vector<unsigned char> tmp_buf(buf_size);
117  ::memcpy(&tmp_buf[0], buf, buf_size);
118 
119 
120  rsc_sparse_vector_u32 csv3; // target vector
121 
122  int res = bm::sparse_vector_deserialize(csv3, &tmp_buf[0], tb);
123  if (res != 0)
124  {
125  std::cerr << "De-Serialization error!" << std::endl;
126  return 1;
127  }
128  if (!csv3.equal(csv2) )
129  {
130  cerr << "Error! Please report a bug to BitMagic project support." << endl;
131  return 1;
132  }
133 
134  // unload rsc to sv, this makes a round-trip to an editable form
135  sparse_vector_u32 sv3(bm::use_null);
136  csv3.load_to(sv3);
137 
138  if (!sv3.equal(sv1) )
139  {
140  std::cerr << "Error! Please report a bug to BitMagic project support." << std::endl;
141  return 1;
142  }
143 
144  print_svector(sv1, true); // print sparse vector again
145 
146  }
147  catch(std::exception& ex)
148  {
149  std::cerr << ex.what() << std::endl;
150  return 1;
151  }

153 
154
