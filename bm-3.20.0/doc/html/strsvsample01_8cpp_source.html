30 #include <iostream>

35 using namespace std;
36 
37 typedef bm::bvector<> bvector_type;
38 
39 // define the sparse vector type for 'char' type using bvector as
40 // a container of bits for bit-transposed planes
41 // 32 - is maximum string length for this container.
42 // Memory allocation is dynamic using sparse techniques, so this number
43 // just defines the max capacity.
44 //
45 typedef bm::str_sparse_vector<char, bvector_type, 32> str_sv_type;
46 
47 int main(void)
48 {
49  try
50  {
51  str_sv_type str_sv;
52 
53  const char* s0 = "asz1234";
54  std::string str1 = "aqw1234";
55  std::string str3 = "54z";
56 
57  str_sv.set(0, s0); // set a C-string
58  str_sv.push_back(str1); // add from an STL string
59 
60  str_sv.assign(3, str3); // assign element 3 from an STL string
61 
62  // please note that container automatically resizes
63  std::cout << "sv size()=" << str_sv.size() << endl; // 4
64 
65  // optimize runs compression in all bit-plains
66  {
67  BM_DECLARE_TEMP_BLOCK(tb)
68  str_sv.optimize(tb);
69  }
70 
71  // print out the container content using [] reference
72  //
73  for (str_sv_type::size_type i = 0; i < str_sv.size(); ++i)
74  {
75  const char* s = str_sv[i];
76  cout << i << ":" << s << endl;
77  } // for i
78  cout << endl;
79 
80  // print content using const_iterator
81  //
82  // const iterator has a bulk implementation,
83  // not like random access const_iterator is transposing a whole set
84  // of elements at once, it is faster for bulk traverse
85  //
86  {
87  str_sv_type::const_iterator it = str_sv.begin();
88  str_sv_type::const_iterator it_end = str_sv.end();
89  for (; it != it_end; ++it)
90  {
91  cout << *it << endl;
92  } // for it
93  }
94  }
95  catch(std::exception& ex)
96  {
97  std::cerr << ex.what() << std::endl;
98  return 1;
99  }
100 
101 
102  return 0;
103 }
