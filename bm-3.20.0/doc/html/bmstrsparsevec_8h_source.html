456  */

458  void assign(size_type idx, const StrType& str)
459  {
460  if (idx >= this->size())
461  this->size_ = idx+1;
462 
463  size_type str_size = size_type(str.size());
464  size_type sz = size_type((str_size < MAX_STR_SIZE) ? str_size : MAX_STR_SIZE-1);
465  if (!sz)
466  {
467  this->clear_value_plains_from(0, idx);
468  return;
469  }
470  unsigned i = 0;
471  for (; i < sz; ++i)
472  {
473  CharType ch = str[i];
474  if (remap_flags_) // compressional re-mapping is in effect
475  {
476  unsigned char remap_value = remap_matrix2_.get(i, unsigned(ch));
477  BM_ASSERT(remap_value);

479  }
480  this->bmatr_.set_octet(idx, i, (unsigned char)ch);
481  if (!ch)
482  break;
483  } // for i
484  if (idx > sz)
485  return;
486  this->bmatr_.set_octet(idx, sz, 0);
487  this->clear_value_plains_from(sz*8+1, idx);

929  break;
930  }

934  BM_ASSERT(remap_value);

936  throw_bad_value(0);
937  str[i] = CharType(remap_value);

942  } // for j

961  str[i] ^= mask;
962  }
963  } // for j

965  {
966  unsigned plain = i*8 + bi;
967  bvector_type* bv = this->bmatr_.get_row(plain);
968  if (!bv)
969  {
970  bv = this->bmatr_.construct_row(plain);
971  bv->init();
972  }
973  bv->set(&bit_list[0], n_bits, BM_SORTED);
974  }
975  } // for k
976  } // for i
977 
978  size_type idx_to = idx_from + imp_size - 1;
979  bvector_type* bv_null = this->get_null_bvect();
980  if (bv_null)
981  bv_null->set_range(idx_from, idx_to);
