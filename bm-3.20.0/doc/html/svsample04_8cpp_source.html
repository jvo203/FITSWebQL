35 static
36 void print_svector(const bm::sparse_vector<unsigned, bm::bvector<> >& sv)
37 {
38  if (sv.size() == 0)
39  {
40  cout << sv.size() << ": [ EMPTY ]" << endl;
41  return;
42  }
43  cout << sv.size() << ": [ ";
44  for (unsigned i = 0; i < sv.size(); ++i)
45  {
46  unsigned v = sv.at(i);
47  bool is_null = sv.is_null(i);
48 
49  if (is_null)
50  cout << "NULL";
51  else
52  cout << v << "";
53 
54  if (i == sv.size()-1)
55  cout << " ]";
56  else
57  cout << ", ";
58  }
59  cout << endl;
60 }

62 int main(void)
63 {
64  try
65  {
66  // sparse vector can support unassigned value semantics
67  // (also known as NULL value in databases)
68  //
69  // to enable this function sparse_vector<> needs to be constructed using
70  // special flag ( bm::use_null )
71  //
72  bm::sparse_vector<unsigned, bm::bvector<> > sv1(bm::use_null);
73 
74  sv1.resize(10);
75  sv1.set(2, 25);
76  sv1.set(3, 35);
77  sv1.set(7, 75);
78 
79  print_svector(sv1); // 10: [ NULL, NULL, 25, 35, NULL, NULL, NULL, 75, NULL, NULL ]
80 
81  sv1.set_null(7);
82 
83  print_svector(sv1); // 10: [ NULL, NULL, 25, 35, NULL, NULL, NULL, NULL, NULL, NULL ]
84 
85  unsigned arr[3] = {1,2,3};
86  sv1.import(arr, 3, 9); // import from a C-style array (by index 9)
87  print_svector(sv1); // 12: [ NULL, NULL, 25, 35, NULL, NULL, NULL, NULL, NULL, 1, 2, 3 ]
88 
89  sv1.clear(2, true); // clear element and set to NULL
90  print_svector(sv1); // 12: [ NULL, NULL, NULL, 35, NULL, NULL, NULL, NULL, NULL, 1, 2, 3 ]
91 
92  sv1.clear(2); // clear element (not NULL anymore)
93  print_svector(sv1); // 12: [ NULL, NULL, 0, 35, NULL, NULL, NULL, NULL, NULL, 1, 2, 3 ]
94 
95  sv1.clear(3); // clear element (it stays NOT NULL)
96  print_svector(sv1); // 12: [ NULL, NULL, NULL, 0, NULL, NULL, NULL, NULL, NULL, 1, 2, 3 ]
97 
98  sv1.clear();
99  print_svector(sv1); // 0: [ EMPTY ]
100 
101  sv1.resize(3);
102  print_svector(sv1); // 3: [ NULL, NULL, NULL ]
103 
104  sv1.push_back(10);
105  sv1.push_back(20);
106 
107  print_svector(sv1); // 5: [ NULL, NULL, NULL, 10, 20 ]
108 
109 
110  // construct not-null-able sparse vector
111  //
112  bm::sparse_vector<unsigned, bm::bvector<> > sv2;
113  sv2.push_back(100);
114  sv2.push_back(200);
115 
116  sv1.join(sv2); // merge two vectors together
117 
118  print_svector(sv1); // 5: [ 100, 200, NULL, 10, 20 ]
119 
120  // construct another NULL-able vector for another join
121  bm::sparse_vector<unsigned, bm::bvector<> > sv3(bm::use_null);
122  sv3.resize(9);
123  sv3.push_back(300);
124 
125  // this join will fully respect the NULL flags assigned in both sparse vectors
126  sv1.join(sv3);
127 
128  print_svector(sv1); // 10: [ 100, 200, NULL, 10, 20, NULL, NULL, NULL, NULL, 300 ]
129 
130  // traverse and print the non-NULL values
131  // using bit-vector of assigned values
132  //
133  // [0] = 100, [1] = 200, [3] = 10, [4] = 20, [9] = 300
134  //
135  {
136  const bm::bvector<>* bv_non_null = sv1.get_null_bvector();
137  if (bv_non_null)
138  {
139  bm::bvector<>::enumerator en = bv_non_null->first();
140 
141  for (; en.valid(); ++en)
142  {
143  unsigned idx = *en;
144  unsigned v = sv1[idx];
145  std::cout << "[" << idx << "] = " << v << ", ";
146  }
147  std::cout << endl;
148  }
149  }
150 
151 
152  // decode sparse_vector with NULL values
153  // decode ignores NULLs (in our case replaces values with '0s'
154  //
155  {
156  std::vector<unsigned> v1(sv1.size());
157  sv1.decode(&v1[0], 0, sv1.size()); // extract elements starting from 0
158 
159  // 100,200,0,10,20,0,0,0,0,300,
160  for (unsigned i = 0; i < v1.size(); ++i)
161  {
162  cout << v1[i] << ",";
163  }
164  cout << endl;
165  }
166  }
167  catch(std::exception& ex)
168  {
169  std::cerr << ex.what() << std::endl;
170  return 1;
171  }
172 
173  return 0;
174 }
