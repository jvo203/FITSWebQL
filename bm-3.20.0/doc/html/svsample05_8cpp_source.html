44 static
45 void print_svector(const sparse_vector_u32& sv)
46 {
47  if (sv.size() == 0)
48  {
49  cout << sv.size() << ": [ EMPTY ]" << endl;
50  return;
51  }
52  cout << sv.size() << ": [ ";
53  for (unsigned i = 0; i < sv.size(); ++i)
54  {
55  unsigned v = sv.at(i);
56  bool is_null = sv.is_null(i);
57 
58  if (is_null)
59  cout << "NULL";
60  else
61  cout << v << "";
62 
63  if (i == sv.size()-1)
64  cout << " ]";
65  else
66  cout << ", ";
67  }
68  cout << endl;
69 }

84 int main(void)
85 {
86  try
87  {
88  // initialize sparse_vector as a binary relation (transform function)
89  // to translate from one set to another (please note it uses NULL values)
90  //
91  sparse_vector_u32 sv_transform(bm::use_null);
92 
93  // transformation function defined as a table:
94  // 2 -> 25
95  // 3 -> 35
96  // 7 -> 75
97  // 1000 -> 2000
98  // 256 -> 2001
99 
100  sv_transform.set(2, 25);
101  sv_transform.set(3, 35);
102  sv_transform.set(7, 75);
103  sv_transform.set(1000, 2000);
104  sv_transform.set(256, 2001);
105 
106  print_svector(sv_transform);
107 
108  // initialize input bit-vector
109  //
110  bm::bvector<> bv_in { 1, 2, 3, 1000, 256 };
111 
112  bm::bvector<> bv_out;
113 
114  bm::set2set_11_transform<sparse_vector_u32> func;
115  func.run(bv_in, sv_transform, bv_out);
116 
117  std::cout << "re-mapped vector:" << endl;
118  print_bvector(bv_out); // ( count = 4): [25, 35, 2000, 2001, ]
119 
120  // Please note, that remapping request for "1" cannot be satisfied
121  // as sv_transform vector did not assign this element so it has
122  // special unassigned, NULL value
123 
124  bm::id_t from = 1;
125  bm::id_t to;
126  bool found = func.remap(1, sv_transform, to);
127  if (!found)
128  std::cout << from << " not mapped" << std::endl; // THIS works!
129  else
130  std::cout << from << " mapped to " << to << std::endl;
131 
132 
133  // now lets try another experiment here, construct image transform
134  // function as default NOT using NULL values
135 
136  sparse_vector_u32 sv_transform2;
137 
138  sv_transform2.set(2, 25);
139  sv_transform2.set(3, 35);
140  sv_transform2.set(7, 75);
141  sv_transform2.set(1000, 2000);
142  sv_transform2.set(256, 2001);
143 
144  found = func.remap(1, sv_transform2, to);
145  if (!found)
146  std::cout << from << " not mapped" << std::endl;
147  else
148  std::cout << from << " mapped to " << to << std::endl; // Now THIS works!
149 
150  // Please note that remapping for unassigned value - works and maps to 0
151  // because transformation sparse vector does not keep information about
152  // unassigned values, all unassigned are now implicitly assumed to be "0"
153 
154  func.run(bv_in, sv_transform2, bv_out);
155 
156  std::cout << "re-mapped vector2:" << endl;
157  print_bvector(bv_out); // ( count = 5): [0, 25, 35, 2000, 2001, ]
158 
159  }
160  catch(std::exception& ex)
161  {

163  return 1;
164  }
165
