44 const unsigned MAX_VALUE = 1000000;
45 
46 static
47 void fill_bvector(bm::bvector<>* bv)
48 {
49  for (unsigned i = 0; i < MAX_VALUE; ++i)
50  {
51  if ((rand() % 10))
52  {
53  bv->set(i);
54  }
55  }
56 }

59 int main(void)
60 {
61  try
62  {
63  bm::bvector<> bv1;
64  bm::bvector<> bv2;
65 
66  fill_bvector(&bv1);
67  fill_bvector(&bv2);
68 
69  cout << "bv1 count = " << bv1.count() << endl;
70  cout << "bv2 count = " << bv2.count() << endl;

73  // Prepare a serializer class
74  // for best performance - create serilizer once and reuse it
75  //
76  BM_DECLARE_TEMP_BLOCK(tb)
77  bm::serializer<bm::bvector<> > bvs(tb);
78  bvs.set_compression_level(4);
79 
80  bm::bvector<>::statistics st1;
81  bm::bvector<>::statistics st2;
82 
83  // compress bit-vectors and compute statistics
84  // (later re-used in serialization)
85  //
86  bv1.optimize(tb, bm::bvector<>::opt_compress, &st1);
87  bv1.optimize(tb, bm::bvector<>::opt_compress, &st2);
88

91  bm::serializer<bm::bvector<> >::buffer sbuf2;
92 
93  // perform serialization
94  //
95  bvs.serialize(bv1, sbuf1, &st1);
96  bvs.serialize(bv2, sbuf2, &st2);
97 
98 
99  // Serialized bvectors (sbuf1 and sbuf2) now ready to be
100  // saved to a database, file or send over a network.
101  // to simulate this we just copy content to std::vector<>
102  //
103 
104  std::vector<unsigned char> vect1;

106 
107  vect1.resize(sbuf1.size());
108  vect2.resize(sbuf2.size());
109

111  ::memcpy(vect2.data(), sbuf2.buf(), sbuf2.size());
112 
113 
114  // Simple deserialization.
115  //
116  bm::bvector<> bv3;
117 
118  // As a result of desrialization bv3 will contain all bits from
119  // bv1 and bv3:
120  // bv3 = bv1 OR bv2
121 
122  bm::deserialize(bv3, sbuf1.buf());
123  bm::deserialize(bv3, sbuf2.buf());
124 
125  bv3.optimize(tb);
126 
127  // A few examples of operation deserializations
128  // set algebraic operation over bit-vector and a BLOB
129  //
130  bm::bvector<> bv4(bv3);
131 
132  // bv4 = (bv1 OR bv2) AND bv1
133  // this must be equal to bv1 ?
134  //
135  bm::operation_deserializer<bm::bvector<> >::deserialize(bv4,
136  vect1.data(),
137  tb,
138  bm::set_AND);
139 
140  cout << "bv4 count = " << bv4.count() << endl;
141  int cmp = bv1.compare(bv4);
142  if (cmp != 0)
143  {
144  cerr << "Logical error detected!" << endl;
145  return 1;
146  }
147  else
148  cout << "bv4 is equal to bv1" << endl;
149 
150  bm::bvector<> bv5(bv3);
151 
152 
153  // if we need just set count, we can get it faster
154  // via set_COUNT_ operations
155  // use of COUNT operations does not materialize a target vector
156  //
157  // POPCNT((bv1 OR bv2) MINUS bv1)
158  unsigned cnt_sub =
159  bm::operation_deserializer<bm::bvector<> >::deserialize(bv3,
160  sbuf1.buf(),
161  tb,
162  bm::set_COUNT_SUB_AB);
163  cout << "minus count = " << cnt_sub << endl;
164 
165 
166  // or we can actually perform the operation and get the full vector
167  // bv5 = (bv1 OR bv2) MINUS bv1
168  //
169  bm::operation_deserializer<bm::bvector<> >::deserialize(bv5,
170  sbuf1.buf(),
171  tb,
172  bm::set_SUB);
173  unsigned bv5_cnt = bv5.count();
174  cout << "bv5 count = " << bv5_cnt << endl;
175 
176  if (cnt_sub != bv5_cnt)
177  {
178  cerr << "Logical error!" << endl;
179  return 1;
180  }
181

183  }
